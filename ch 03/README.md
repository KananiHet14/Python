these all are usefull string functions

strip()        # remove spaces both sides
lstrip()       # remove left spaces
rstrip()       # remove right spaces
lower()        # convert to lowercase
upper()        # convert to uppercase
title()        # first letter capital
replace()      # replace substring
split()        # split into list
join()         # join list into string
find()         # find position of substring
count()        # count occurrences
startswith()   # check starting text
endswith()     # check ending text
isdigit()      # check if all digits
isalpha()      # check if all letters
isalnum()      # check letters and digits
format()       # insert values into string
zfill()        # fill zeros on left
len()          # total characters
