these are all usefull List and Tuple methods

LIST{
        append()      # add item at end
        extend()      # add multiple items
        insert()      # add at specific index
        remove()      # remove by value
        pop()         # remove by index
        sort()        # sort list
        reverse()     # reverse list
        index()       # find position of item
        count()       # count occurrences
        copy()        # copy list
        clear()       # empty the list
        len()         # total items
}

TUPLE{

        count()       # count occurrences
        index()       # find position
        len()         # total items
        sorted()      # returns sorted list from tuple
        min()         # smallest value
        max()         # largest value

}
