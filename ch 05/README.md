these all are Dictionary and Sets methods


DICTIONARY {
        get()          # get value safely
        keys()         # all keys
        values()       # all values
        items()        # all key-value pairs
        update()       # add or update keys
        pop()          # remove a key
        setdefault()   # get key or set default
        copy()         # copy dictionary
        clear()        # empty dictionary
        len()          # total keys
        in             # check if key exists
}


SET {
        add()                # add one item
        update()             # add multiple items
        remove()             # remove item (error if not found)
        discard()            # remove item (no error)
        pop()                # remove random item
        union()              # combine two sets
        intersection()       # common items only
        difference()         # items in one not other
        issubset()           # check if subset
        issuperset()         # check if superset
        clear()              # empty the set
}
